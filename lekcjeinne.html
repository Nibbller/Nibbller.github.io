Napisz program obliczający obwód i pole trapezu. 

#include <iostream>
    using namespace std;
        
    int main()
    {
        float a,b,h;
        cout << "Podaj a: ";
        cin >> a;
        cout << "Podaj b: ";
        cin >> b;
        cout << "Podaj h: ";
        cin >> h;
        
        cout << "Obwod = " << a+b+2*h << endl;
        cout << "Pole = " << (a+b)*h/2;
        
        return 0;
    }
Napisz program obliczający pierwiastek sześcienny z liczby podanej przez użytkownika. 

#include <iostream>
    #include <cmath>
    using namespace std;
        
    int main()
    {
        int l1;
        cout << "Podaj pierwsza liczbe: ";
        cin >> l1;
        
        cout << "wynik = " << pow(l1,1/3.0);
        
        return 0;
    }
